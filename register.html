function register() { const name = document.getElementById("name").value.trim(); const email = document.getElementById("email").value.trim(); const password = document.getElementById("password").value.trim(); const message = document.getElementById("message"); if (!name || !email || !password) { message.textContent = "Please fill out all fields."; message.className = "message error"; return; } auth.createUserWithEmailAndPassword(email, password) .then((userCredential) => { message.textContent = "Account created successfully!"; message.className = "message success"; console.log("User registered:", userCredential.user); // Update display name userCredential.user.updateProfile({ displayName: name }) .then(() => { // Redirect to dashboard after profile update window.location.href = "dashboard.html"; }); }) .catch((error) => { message.textContent = error.message; message.className = "message error"; console.error("Registration error:", error); }); }
